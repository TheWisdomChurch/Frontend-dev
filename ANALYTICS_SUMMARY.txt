🚀 QUICK START (10 MINUTES)
─────────────────────────────────────────────────────────────────────────────

1. Update src/app/layout.tsx:
   
   import AnalyticsProviderV2 from '@/components/providers/AnalyticsProviderV2';
   
   export default function RootLayout({ children }) {
     return (
       <html>
         <body>
           <AnalyticsProviderV2>
             {children}
           </AnalyticsProviderV2>
         </body>
       </html>
     );
   }

2. Add to .env.local:
   
   NEXT_PUBLIC_ANALYTICS_API=/api/analytics

3. Start tracking in components:
   
   'use client';
   import { usePageView, useClickTracking } from '@/hooks/useAnalyticsV2';
   
   export default function MyPage() {
     usePageView('My Page');
     const clickRef = useClickTracking('My Button');
     return <button ref={clickRef}>Click Me</button>;
   }

4. Test in development:
   
   curl http://localhost:3000/api/analytics

💻 REAL-WORLD EXAMPLES
─────────────────────────────────────────────────────────────────────────────

1. Track Homepage:
   usePageView('Home', 'https://example.com');

2. Track Donation Button:
   const clickRef = useClickTracking('Donate Button');
   <button ref={clickRef}>Donate</button>

3. Track Form Submission:
   const { trackSubmit } = useFormTracking('Contact');
   const handleSubmit = () => trackSubmit({ name, email });

4. Track Purchase:
   const { trackPurchase } = useConversionTracking();
   trackPurchase('order-123', 100, 'USD');

5. Track User Login:
   const { identifyUser } = useUserIdentification();
   identifyUser('user-123', { email: 'user@example.com' });

6. Track Video Playback:
   const { trackPlay, trackComplete } = useMediaTracking('Service Video');
   <video onPlay={() => trackPlay()} onEnded={() => trackComplete()}>

7. Get Engagement Metrics:
   const { getEngagementScore, getMetrics } = useEngagementTracking();
   console.log(getEngagementScore()); // 0-100
